## §2 核心设计（One Track 零 break change 原则）

### §2.1 FSMState 枚举扩展（One Track：**旧值 0~8 冻结，仅末尾追加 9/10**）

```gdscript
# 追加到 scripts/editor/CharacterBase.gd FSMState enum 末尾（原 DEAD=8 → 下面加 2 行，不改旧值）
enum FSMState {
	IDLE,      # 0  （V0.3c 旧值不动）
	RUN,       # 1
	JUMP,      # 2
	ATTACK1,   # 3
	ATTACK2,   # 4
	ATTACK3,   # 5
	HURT,      # 6
	BLOCK,     # 7
	DEAD,      # 8
	DOUBLEJUMP,# 9  ← V0.3d 新增：二段跳上升/下落（玩家从地面 JUMP 后再按 Space 触发）
	DASH       # 10 ← V0.3d 新增：冲刺位移 + 无敌帧（LeftShift 触发；0.7s CD；消耗 20 stamina）
}
```

### §2.1.1 合法跳转表扩展（DOUBLEJUMP/DASH 加入原表末尾，不重排数组）

原 `_legal_transition` 字典保持 From 索引不动，仅：
- 允许 `JUMP(2)` → DOUBLEJUMP(9)；允许 `DOUBLEJUMP(9)` → IDLE/RUN/HURT/DEAD
- 允许 `IDLE/RUN/JUMP/BLOCK` → DASH(10)；允许 `DASH(10)` → IDLE/RUN/JUMP/HURT/DEAD

### §2.2 InputBus 信号追加（旧信号 0 改动：首字母大写继续保留）

```gdscript
# autoload/InputBus.gd 原有信号完全保留（JumpPressed/AttackPressed/BlockPressed…）
# 在 signal 区域末尾追加 2 条新信号，不改动旧信号顺序/参数
signal axis_changed(h_axis: float, v_axis: float)  # V0.3d 新增：每帧 Process 里 moveAxis 变化触发（默认 v_axis=0）
signal weapon_changed(slot: int)                   # V0.3d 新增：1/2/3 键按下触发 slot=1/2/3

# _unhandled_input 末尾追加：1/2/3 键 = 切武器
func _unhandled_input(event: InputEvent) -> void:
	# 【原有旧代码一字不动：JumpPressed/AttackPressed 等】
	#  ↓↓↓ V0.3d 新增 ↓↓↓
	if event.is_action_pressed("weapon_1"): weapon_changed.emit(1)
	if event.is_action_pressed("weapon_2"): weapon_changed.emit(2)
	if event.is_action_pressed("weapon_3"): weapon_changed.emit(3)

# _process 末尾追加：每帧 axis 变化 emit axis_changed
func _process(_delta: float) -> void:
	# 【原有旧代码一字不动：moveAxis / blockStrength 边沿检测】
	#  ↓↓↓ V0.3d 新增：有变化才 emit（避免无意义信号风暴）
	var v_axis: float = 0.0  # 当前版本未实现上下移动（Y轴攻击锁定），保留 v 参数供 V0.4 扩展
	var new_h: float = moveAxis
	if abs(new_h - _last_axis_h) > 0.001:
		axis_changed.emit(new_h, v_axis)
		_last_axis_h = new_h
```

> **One Track 兼容验证**：旧 PlayerBase 订阅的 `InputBus.jump_pressed`（小写旧名）V0.3d 会**断开无效** → 所以 PlayerBase 需改成订阅真实大写名 `InputBus.JumpPressed`（这就是本迭代 D4 修改的任务，对齐后新旧都没问题）。

#### 改造点 C：攻击命中走 CDC 7 步 + take_damage(opts._use_cdc=true)（V0.3b 流水线真实跑，0 桩）
```gdscript
# PlayerBase.gd 原 take_damage_delegate 替换为真实 CDC 7 步
func _do_attack_hit(cfg: Dictionary) -> void:
	attack_hit_done = true
	var atk_val: int = base_atk
	var atk_mult: float = float(cfg.get("atk_mult", 1.0))
	if typeof(weapon) == TYPE_DICTIONARY and weapon.has("atk_mult"):
		atk_mult *= float(weapon.atk_mult)
	var range_px: float = 40.0
	if typeof(weapon) == TYPE_DICTIONARY:
		range_px = float(weapon.get("range", 40.0))
	var is_shield_break: bool = false
	if typeof(weapon) == TYPE_DICTIONARY:
		is_shield_break = bool(weapon.get("break_shield", false))
	var hit_origin: Vector2 = global_position + Vector2(range_px * 0.5 * facing, 0.0)
	var targets: Array[Node] = _collect_enemies_in_range(hit_origin, range_px)
	for t in targets:
		if t == self or not is_instance_valid(t):
			continue
		var atkr_dict: Dictionary = {
			"atk": atk_val,
			"atk_mult": atk_mult,
			"facing": facing,
			"kind": kind,
			"weapon_id": current_weapon_id,
			"break_shield": is_shield_break,
			"crit_rate_bonus": 0.0
		}
		var ctx_dict: Dictionary = {
			"attack_angle_rad": 0.0,
			"damage_type": 0  # 0 = PHYSICAL（对齐 V0.3a DamageType.PHYSICAL 枚举）
		}
		var opts: Dictionary = {
			"_use_cdc": true,
			"attacker_dict": atkr_dict,
			"context_dict": ctx_dict,
			"hitstun": 0.18,
			"knockback": 60.0,
			"ignore_block": is_shield_break
		}
		# t 是真实 CharacterBase（稻草人/敌人），走 V0.3c take_damage → CDC 7 步 → FSM 自动切 HURT
		var final_dmg: int = t.take_damage(0, self, opts)
		if final_dmg > 0 and _autoload("GameEvents") != null:
			var ge: Node = _autoload("GameEvents")
			ge.emit_signal("character_attack_connected", self, t, final_dmg)
			ge.emit_signal("damage_calculated", atkr_dict, {"def": base_def}, ctx_dict, {"final_damage": final_dmg})
```

### §2.5 FSM 状态切换窗口时序（T4 测试用例对齐：连击窗口 0.55s，超时自动回 IDLE）

| 事件 | 时序（秒） | FSM 行为 |
|------|-----------|---------|
| 第 1 次 J 按下 | t=0.00s | state→ATTACK1；combo_window_left=0.55s；state_timer=0 |
| ATTACK1 总时长 | windup0.08 + active0.15 + recovery0.20 = **0.43s** | t=0.08~0.23s = active_window（命中检测开 1 次）；t=0.43s ATTACK1 结束 |
| 第 2 次 J 按下（窗口内） | t=0.30s（<0.55） | combo_index=2；state→ATTACK2；combo_window_left 重置 0.55s |
| 第 2 次 J 按下（超时） | t=0.90s（>0.55） | combo_index 重置为 1；state→ATTACK1（重新开始） |
