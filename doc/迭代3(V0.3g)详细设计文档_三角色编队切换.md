## 三、PartyManager 核心设计（Autoload单例）

```gdscript
# autoload/PartyManager.gd
extends Node
class_name PartyManager

var members: Array[CharacterBody2D] = []
var active_idx: int = 0
signal party_changed
signal party_switched(old_idx: int, new_idx: int, new_char: CharacterBody2D)

func register(ch: CharacterBody2D) -> void:  # 加角色到队伍
    if members.size() >= 3: return
    if not members.has(ch): members.append(ch)
    if members.size() == 1: _activate(0)
    party_changed.emit()

func switch_next() -> int:                      # Tab: 循环下一人
    if members.size() == 0: return -1
    var ni := (active_idx + 1) % members.size()
    return _activate(ni)

func switch_to(idx: int) -> int:                 # F1/F2/F3: 选指定人
    if idx < 0 or idx >= members.size(): return -1
    return _activate(idx)

func _activate(idx: int) -> int:
    var old: int = active_idx
    active_idx = idx
    var nc: CharacterBody2D = members[idx]
    party_switched.emit(old, idx, nc)
    party_changed.emit()
    return idx

func get_active() -> CharacterBody2D:
    if members.is_empty(): return null
    return members[active_idx]
```

**依赖合规（单向）**：PartyManager不依赖GameEvents/InputBus（它是被调用方，上层调用）

## 八、10UC 无桩E2E测试用例（无场景树依赖，RefCounted字典模拟）

| UC | 用例名 | 断言要点 |
|----|--------|---------|
| T1 | register() 3人 → members.size()=3 | 数组长度=3, active_idx初始=0 |
| T2 | register() 4th 人 → 失败忽略(上限3) | members.size()仍=3 |
| T3 | 3角色max_hp/atk/形象各不相同 | Mace>=140, Spear<=90, Farmer=100 |
| T4 | switch_next() 0→1→2→0 循环 | 每次返回新idx |
| T5 | switch_to(2)枪兵 → get_active()=3号位 | active_idx=2 |
| T6 | switch_to(-1)越界 → 返回-1,active_idx不变 | 边界保护 |
| T7 | 切换后 party_switched 信号触发次数+payload | 信号参数old/new匹配 |
| T8 | 2次switch_next → 与switch_to((0+2)%size)等价 | 纯逻辑一致性 |
| T9 | 同实例重复register → members不重复 | 去重逻辑 |
| T10 | get_active()未注册→null，注册后→第1个 | 初始化边界 |
